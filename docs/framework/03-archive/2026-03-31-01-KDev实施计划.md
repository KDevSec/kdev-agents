**Step 4: 创建 src/state/state-manager.ts**

```typescript
import * as fs from 'fs';
import * as path from 'path';
import type { GateReport, GateName } from '../types';
import type { KDevState, Phase } from './types';
import { INITIAL_STATE } from './types';

export class StateManager {
  private state: KDevState;
  private stateFile: string;

  constructor(private baseDir: string = '.kdev') {
    this.stateFile = path.join(baseDir, 'state.json');
    this.state = this.load();
  }

  getState(): KDevState {
    return { ...this.state };
  }

  init(feature: string): void {
    this.state = {
      ...INITIAL_STATE,
      sessionId: this.generateSessionId(),
      feature,
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString(),
    };
    this.ensureDir();
    this.save();
  }

  transition(phase: Phase): void {
    if (this.state.currentPhase !== 'DONE') {
      this.state.phases[this.state.currentPhase].status = 'completed';
      this.state.phases[this.state.currentPhase].completedAt = new Date().toISOString();
    }

    this.state.currentPhase = phase;
    this.state.phases[phase].status = 'in_progress';
    this.state.phases[phase].startedAt = new Date().toISOString();
    this.state.updatedAt = new Date().toISOString();
  }

  recordGateResult(gate: GateName, report: GateReport): void {
    this.state.gates[gate] = report;
    this.state.updatedAt = new Date().toISOString();
  }

  save(): void {
    this.ensureDir();
    fs.writeFileSync(this.stateFile, JSON.stringify(this.state, null, 2));
  }

  private load(): KDevState {
    if (fs.existsSync(this.stateFile)) {
      const content = fs.readFileSync(this.stateFile, 'utf-8');
      return JSON.parse(content);
    }
    return { ...INITIAL_STATE };
  }

  private ensureDir(): void {
    if (!fs.existsSync(this.baseDir)) {
      fs.mkdirSync(this.baseDir, { recursive: true });
    }
  }

  private generateSessionId(): string {
    return `kdev-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
  }
}
```

```typescript
import * as fs from 'fs';
import * as path from 'path';
import type { Agent } from '../types';

export class AgentLoader {
  constructor(private agentsDir: string) {}

  load(name: string): Agent | undefined {
    const filePath = path.join(this.agentsDir, `${name}.md`);

    if (!fs.existsSync(filePath)) {
      return undefined;
    }

    const content = fs.readFileSync(filePath, 'utf-8');
    return this.parseAgent(name, content);
  }

  loadAll(): Agent[] {
    const agents: Agent[] = [];
    const files = fs.readdirSync(this.agentsDir);

    for (const file of files) {
      if (file.endsWith('.md')) {
        const name = path.basename(file, '.md');
        const agent = this.load(name);
        if (agent) {
          agents.push(agent);
        }
      }
    }

    return agents;
  }

  private parseAgent(name: string, content: string): Agent {
    const frontmatterMatch = content.match(/^---\n([\s\S]*?)\n---/);

    if (!frontmatterMatch) {
      throw new Error(`Invalid agent file format: ${name}`);
    }

    const frontmatter = frontmatterMatch[1];
    const metadata = this.parseFrontmatter(frontmatter);

    return {
      name: metadata.name || name,
      model: metadata.model || 'sonnet',
      role: this.inferRole(name),
      tools: this.extractTools(content),
    };
  }

  private parseFrontmatter(text: string): Record<string, string> {
    const result: Record<string, string> = {};
    const lines = text.split('\n');

    for (const line of lines) {
      const [key, ...values] = line.split(':');
      if (key && values.length > 0) {
        result[key.trim()] = values.join(':').trim();
      }
    }

    return result;
  }

  private inferRole(name: string): 'planning' | 'implementation' | 'review' | 'testing' | 'exploration' {
    const roleMap: Record<string, 'planning' | 'implementation' | 'review' | 'testing' | 'exploration'> = {
      planner: 'planning',
      executor: 'implementation',
      reviewer: 'review',
      tester: 'testing',
      explorer: 'exploration',
    };
    return roleMap[name] || 'implementation';
  }

  private extractTools(content: string): string[] {
    const toolsMatch = content.match(/## Tools Available\n([\s\S]*?)(?:\n##|$)/);
    if (!toolsMatch) return [];

    const toolsText = toolsMatch[1];
    const tools: string[] = [];

    const toolPatterns = [
      'Read', 'Write', 'Edit', 'Bash', 'Glob', 'Grep', 'LSP',
      'AskUserQuestion', 'Playwright MCP'
    ];

    for (const tool of toolPatterns) {
      if (toolsText.includes(tool)) {
        tools.push(tool);
      }
    }

    return tools;
  }
}
```

```typescript
import type {
  UnifiedSpec,
  UnifiedPlan,
  PlanTask,
  ExecutionStep,
  FeatureSpec,
  StorySpec,
  AcceptanceCriteria,
  TaskDependency,
} from '../types';

/**
 * 规格转换器
 *
 * 职责：
 * 1. 将 spec-kit 格式转换为 UnifiedSpec
 * 2. 将 OpenSpec 格式转换为 UnifiedSpec
 * 3. 将 UnifiedSpec 转换为 UnifiedPlan（供 writing-plans 消费）
 */
export class SpecTransformer {
  /**
   * 从 spec-kit 格式转换
   */
  fromSpecKit(raw: Record<string, unknown>): UnifiedSpec {
    const project = raw.project as Record<string, unknown> || {};
    const constitution = raw.constitution as Record<string, unknown> || {};
    const specs = raw.specs as Array<Record<string, unknown>> || [];

    return {
      id: `spec-${Date.now()}`,
      version: '1.0.0',
      source: 'spec-kit',
      createdAt: new Date().toISOString(),
      project: {
        name: (project.name as string) || 'unknown',
        type: this.inferProjectType(raw),
        techStack: (raw.techStack as UnifiedSpec['project']['techStack']) || {},
      },
      constitution: {
        principles: (constitution.principles as string[]) || [],
        conventions: (constitution.conventions as Record<string, string[]>) || {},
        constraints: (constitution.constraints as string[]) || [],
      },
      features: specs.map(s => this.transformSpecKitFeature(s)),
      status: 'draft',
    };
  }

  /**
   * 从 OpenSpec 格式转换
   */
  fromOpenSpec(raw: Record<string, unknown>): UnifiedSpec {
    const changes = raw.changes as Array<Record<string, unknown>> || [];

    return {
      id: `spec-${Date.now()}`,
      version: '1.0.0',
      source: 'openspec',
      createdAt: new Date().toISOString(),
      project: {
        name: (raw.project as string) || 'unknown',
        type: 'web',
        techStack: {},
      },
      features: changes.map(c => this.transformOpenSpecChange(c)),
      status: 'draft',
    };
  }

  /**
   * 将 UnifiedSpec 转换为 UnifiedPlan
   *
   * 这是关键的衔接点：
   * - 消费 UnifiedSpec（来自 spec-kit / OpenSpec）
   * - 输出 UnifiedPlan（供 superpowers:writing-plans 消费）
   */
  toPlan(spec: UnifiedSpec): UnifiedPlan {
    const tasks: PlanTask[] = [];
    const dependencies: TaskDependency[] = [];

    // 为每个 AR 生成 Task
    for (const feature of spec.features) {
      for (const story of feature.stories) {
        for (const ac of story.acceptanceCriteria) {
          const task = this.createTaskFromAC(ac, story, feature, spec.project);
          tasks.push(task);
        }
      }
    }

    // 推断依赖关系
    this.inferDependencies(tasks, dependencies);

    return {
      id: `plan-${Date.now()}`,
      specId: spec.id,
      createdAt: new Date().toISOString(),
      summary: `Implementation plan for ${spec.features.length} features, ${tasks.length} tasks`,
      architecture: this.generateArchitectureSummary(spec),
      tasks,
      dependencies,
    };
  }

  // ===== 私有方法 =====

  private transformSpecKitFeature(raw: Record<string, unknown>): FeatureSpec {
    const stories = (raw.stories as Array<Record<string, unknown>>) || [];

    return {
      id: (raw.id as string) || `FEAT-${Date.now()}`,
      title: (raw.title as string) || 'Untitled',
      description: (raw.description as string) || '',
      priority: (raw.priority as FeatureSpec['priority']) || 'P1',
      stories: stories.map(s => this.transformSpecKitStory(s)),
    };
  }

  private transformSpecKitStory(raw: Record<string, unknown>): StorySpec {
    const acList = (raw.acceptanceCriteria as Array<Record<string, unknown>>) || [];

    return {
      id: (raw.id as string) || `SR-${Date.now()}`,
      title: (raw.title as string) || 'Untitled',
      userStory: (raw.userStory as string) || '',
      acceptanceCriteria: acList.map((ac, idx) => ({
        id: (ac.id as string) || `AR-${raw.id}-${idx + 1}`,
        given: (ac.given as string) || '',
        when: (ac.when as string) || '',
        then: (ac.then as string) || '',
      })),
      estimatedPoints: (raw.estimatedPoints as number) || 3,
      dependencies: (raw.dependencies as string[]) || [],
    };
  }

  private transformOpenSpecChange(raw: Record<string, unknown>): FeatureSpec {
    return {
      id: (raw.id as string) || `FEAT-${Date.now()}`,
      title: (raw.title as string) || 'Untitled',
      description: (raw.description as string) || '',
      priority: 'P1',
      stories: [], // OpenSpec 格式不同，需要额外处理
    };
  }

  private createTaskFromAC(
    ac: AcceptanceCriteria,
    story: StorySpec,
    feature: FeatureSpec,
    project: UnifiedSpec['project']
  ): PlanTask {
    return {
      id: `TASK-${ac.id}`,
      title: `${story.title}: ${ac.given}`,
      description: `Given: ${ac.given}\nWhen: ${ac.when}\nThen: ${ac.then}`,
      priority: feature.priority,
      acceptanceCriteriaId: ac.id,
      files: {
        create: [],
        modify: [],
        test: [],
      },
      steps: this.generateTDDSteps(ac),
      estimatedMinutes: 5,
    };
  }

  private generateTDDSteps(ac: AcceptanceCriteria): ExecutionStep[] {
    return [
      {
        order: 1,
        action: 'write_test',
        description: `Write failing test for: ${ac.then}`,
      },
      {
        order: 2,
        action: 'run_test',
        description: 'Verify test fails (RED)',
        command: 'npm test',
        expectedOutput: 'FAIL',
      },
      {
        order: 3,
        action: 'write_code',
        description: `Implement minimal code for: ${ac.when}`,
      },
      {
        order: 4,
        action: 'run_test',
        description: 'Verify test passes (GREEN)',
        command: 'npm test',
        expectedOutput: 'PASS',
      },
      {
        order: 5,
        action: 'refactor',
        description: 'Refactor if needed',
      },
      {
        order: 6,
        action: 'commit',
        description: 'Commit changes',
      },
    ];
  }

  private inferDependencies(tasks: PlanTask[], dependencies: TaskDependency[]): void {
    // 简单实现：按顺序建立依赖
    for (let i = 1; i < tasks.length; i++) {
      dependencies.push({
        from: tasks[i - 1].id,
        to: tasks[i].id,
        type: 'blocks',
      });
    }
  }

  private generateArchitectureSummary(spec: UnifiedSpec): string {
    const tech = spec.project.techStack;
    const parts: string[] = [];

    if (tech.frontend?.length) {
      parts.push(`Frontend: ${tech.frontend.join(', ')}`);
    }
    if (tech.backend?.length) {
      parts.push(`Backend: ${tech.backend.join(', ')}`);
    }
    if (tech.database?.length) {
      parts.push(`Database: ${tech.database.join(', ')}`);
    }

    return parts.length > 0 ? parts.join(' | ') : 'Architecture TBD';
  }

  private inferProjectType(raw: Record<string, unknown>): UnifiedSpec['project']['type'] {
    const techStack = raw.techStack as Record<string, string[]> || {};

    if (techStack.frontend?.length && techStack.backend?.length) {
      return 'web';
    }
    if (techStack.backend?.length && !techStack.frontend?.length) {
      return 'api';
    }
    return 'web';
  }
}
```

```typescript
/**
 * 技能调用契约
 *
 * 定义每个技能的输入输出格式，确保技能间数据流一致性
 */
export interface SkillContract {
  /** 输入格式：文件路径或数据类型 */
  input: string;
  /** 输出格式：文件路径 */
  output: string;
  /** 输出数据的 schema 名称 */
  schema: string;
  /** 可选的转换函数名称 */
  transform?: string;
  /** 依赖的外部技能 */
  dependsOn?: string[];
}

export const SKILL_CONTRACTS: Record<string, SkillContract> = {
  /**
   * /kdev-spec 调用契约
   *
   * 职责：生成需求规格
   * 调用：spec-kit / OpenSpec
   */
  'kdev-spec': {
    input: 'string', // 用户需求描述
    output: '.kdev/specs/current.json',
    schema: 'UnifiedSpec',
    dependsOn: ['spec-kit', 'openspec'],
  },

  /**
   * /kdev-plan 调用契约
   *
   * 职责：将规格转换为实施计划
   * 调用：superpowers:writing-plans
   */
  'kdev-plan': {
    input: '.kdev/specs/current.json',
    output: '.kdev/plans/current.json',
    schema: 'UnifiedPlan',
    transform: 'SpecTransformer.toPlan()',
    dependsOn: ['superpowers:writing-plans'],
  },

  /**
   * /kdev-exec 调用契约
   *
   * 职责：执行实施计划
   * 调用：superpowers:subagent-driven-development
   */
  'kdev-exec': {
    input: '.kdev/plans/current.json',
    output: '.kdev/sessions/',
    schema: 'ExecutionLog[]',
    dependsOn: ['superpowers:subagent-driven-development', 'superpowers:test-driven-development'],
  },

  /**
   * /kdev-test 调用契约
   *
   * 职责：运行集成测试和E2E测试
   * 调用：playwright-skill
   */
  'kdev-test': {
    input: '.kdev/plans/current.json',
    output: '.kdev/test-reports/',
    schema: 'TestReport',
    dependsOn: ['playwright-skill'],
  },

  /**
   * /kdev-review 调用契约
   *
   * 职责：代码审查
   * 调用：superpowers:requesting-code-review
   */
  'kdev-review': {
    input: '.kdev/plans/current.json',
    output: '.kdev/reviews/current.md',
    schema: 'ReviewReport',
    dependsOn: ['superpowers:requesting-code-review'],
  },

  /**
   * /kdev-verify 调用契约
   *
   * 职责：最终验证
   * 调用：superpowers:verification-before-completion
   */
  'kdev-verify': {
    input: '.kdev/reviews/current.md',
    output: '.kdev/gates/verify.json',
    schema: 'GateReport',
    dependsOn: ['superpowers:verification-before-completion'],
  },
};

/**
 * 验证契约输入是否存在
 */
export function validateContract(skillName: string, inputPath: string): boolean {
  const contract = SKILL_CONTRACTS[skillName];
  if (!contract) {
    return false;
  }

  // 如果输入是数据类型而非文件路径，直接返回 true
  if (!inputPath.startsWith('.') && !inputPath.startsWith('/')) {
    return true;
  }

  // 检查文件是否存在
  return fs.existsSync(inputPath);
}

/**
 * 获取技能的输出路径
 */
export function getOutputPath(skillName: string, baseDir: string = '.kdev'): string {
  const contract = SKILL_CONTRACTS[skillName];
  if (!contract) {
    throw new Error(`Unknown skill: ${skillName}`);
  }

  return path.join(baseDir, contract.output);
}

/**
 * 获取技能依赖链
 */
export function getDependencyChain(skillName: string): string[] {
  const contract = SKILL_CONTRACTS[skillName];
  if (!contract || !contract.dependsOn) {
    return [];
  }

  const chain: string[] = [...contract.dependsOn];

  for (const dep of contract.dependsOn) {
    const depChain = getDependencyChain(dep);
    chain.push(...depChain);
  }

  return [...new Set(chain)]; // 去重
}
```

### Key Design Decisions

#### 1. 技能间数据流衔接

**问题**：spec-kit、OpenSpec 生成的规格格式不一致，无法直接被 superpowers 消费

**解决方案**：

```
spec-kit/OpenSpec  ──▶  UnifiedSpec  ──▶  UnifiedPlan  ──▶  superpowers/OMC
                              │
                              ▼
                        SpecTransformer
```

- **UnifiedSpec**: 需求规格的统一表示
- **UnifiedPlan**: 实施计划的统一表示（包含 TDD 步骤）
- **SpecTransformer**: 负责格式转换

#### 2. TDD 步骤自动生成

每个 Acceptance Criteria 自动生成 TDD 步骤：

```typescript
steps: [
  { action: 'write_test', description: 'Write failing test' },
  { action: 'run_test', expectedOutput: 'FAIL' },
  { action: 'write_code', description: 'Implement minimal code' },
  { action: 'run_test', expectedOutput: 'PASS' },
  { action: 'refactor' },
  { action: 'commit' },
]
```

#### 3. 技能调用契约

每个技能明确定义输入输出：

| 技能 | 输入 | 输出 | Schema |
|------|------|------|--------|
| /kdev-spec | string | .kdev/specs/current.json | UnifiedSpec |
| /kdev-plan | .kdev/specs/current.json | .kdev/plans/current.json | UnifiedPlan |
| /kdev-exec | .kdev/plans/current.json | .kdev/sessions/ | ExecutionLog[] |
