/**
 * kdev-agent 状态类型定义

export type Phase = 'INIT' | 'SPEC' | 'PLAN' | 'EXEC' | 'REVIEW' | 'VERIFY';

export type PhaseStatus = 'PENDING' | 'IN_PROGRESS' | 'COMPLETED' | 'BLOCKED';

export interface GateResult {
  passed: boolean;
  timestamp: string;
  message?: string;
}

export interface PhaseState {
  status: PhaseStatus;
  startedAt?: string;
  completedAt?: string;
  gate?: GateResult;
  confirmation?: {
    required: boolean;
    confirmed: boolean;
    confirmer?: string;
    confirmedAt?: string;
  };
}

export interface KdevState {
  sessionId: string;
  feature: string;
  currentPhase: Phase;
  phases: Record<Phase, PhaseState>;
  createdAt: string;
  updatedAt: string;
}

export const DEFAULT_PHASES: Record<Phase, PhaseState> = {
  INIT: { status: 'PENDING' },
  SPEC: { status: 'PENDING', confirmation: { required: true, confirmed: false } },
  PLAN: { status: 'PENDING', confirmation: { required: false, confirmed: false } },
  EXEC: { status: 'PENDING' },
  REVIEW: { status: 'PENDING', confirmation: { required: true, confirmed: false } },
  VERIFY: { status: 'PENDING' },
};

Create: `src/state/manager.ts`

```typescript
import * as fs from 'fs';
import * as path from 'path';
import { KdevState, Phase, DEFAULT_PHASES } from './types.js';

const STATE_DIR = '.kdev';
const STATE_FILE = 'state.json';

export class StateManager {
  private basePath: string;
  private statePath: string;

  constructor(basePath: string) {
    this.basePath = basePath;
    this.statePath = path.join(basePath, STATE_DIR, STATE_FILE);
  }

  /**
   * 初始化状态
   */
  init(feature: string): KdevState {
    const state: KdevState = {
      sessionId: `kdev-${Date.now()}`,
      feature,
      currentPhase: 'INIT',
      phases: { ...DEFAULT_PHASES },
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString(),
    };

    this.ensureDir();
    this.write(state);
    return state;
  }

  /**
   * 读取状态
   */
  read(): KdevState | null {
    if (!fs.existsSync(this.statePath)) {
      return null;
    }
    const content = fs.readFileSync(this.statePath, 'utf-8');
    return JSON.parse(content);
  }

  /**
   * 写入状态
   */
  write(state: KdevState): void {
    state.updatedAt = new Date().toISOString();
    this.ensureDir();
    fs.writeFileSync(this.statePath, JSON.stringify(state, null, 2));
  }

  /**
   * 更新阶段
   */
  updatePhase(phase: Phase, updates: Partial<KdevState['phases'][Phase]>): KdevState | null {
    const state = this.read();
    if (!state) return null;

    state.phases[phase] = { ...state.phases[phase], ...updates };
    state.currentPhase = phase;
    this.write(state);
    return state;
  }

  /**
   * 确认阶段
   */
  confirmPhase(phase: Phase, confirmer: string): KdevState | null {
    const state = this.read();
    if (!state) return null;

    if (state.phases[phase].confirmation) {
      state.phases[phase].confirmation!.confirmed = true;
      state.phases[phase].confirmation!.confirmer = confirmer;
      state.phases[phase].confirmation!.confirmedAt = new Date().toISOString();
    }
    this.write(state);
    return state;
  }

  private ensureDir(): void {
    const dir = path.dirname(this.statePath);
    if (!fs.existsSync(dir)) {
      fs.mkdirSync(dir, { recursive: true });
    }
  }
}
```

import { McpServer } from '@modelcontextprotocol/sdk/server/mcp.js';
import { z } from 'zod';
import { StateManager } from '../../state/manager.js';

export function registerStateTools(server: McpServer): void {
  // kdev_state_init - 初始化状态
  server.tool(
    'kdev_state_init',
    'Initialize kdev state for a feature',
    {
      feature: z.string().describe('Feature name or description'),
      basePath: z.string().optional().describe('Project base path (defaults to current directory)'),
    },
    async ({ feature, basePath }) => {
      const manager = new StateManager(basePath || process.cwd());
      const state = manager.init(feature);
      return {
        content: [{
          type: 'text',
          text: JSON.stringify(state, null, 2),
        }],
      };
    }
  );

  // kdev_state_read - 读取状态
  server.tool(
    'kdev_state_read',
    'Read current kdev state',
    {
      basePath: z.string().optional().describe('Project base path'),
    },
    async ({ basePath }) => {
      const manager = new StateManager(basePath || process.cwd());
      const state = manager.read();
      if (!state) {
        return {
          content: [{
            type: 'text',
            text: 'No active kdev session found. Run kdev_state_init first.',
          }],
        };
      }
      return {
        content: [{
          type: 'text',
          text: JSON.stringify(state, null, 2),
        }],
      };
    }
  );

  // kdev_state_update - 更新阶段状态
  server.tool(
    'kdev_state_update',
    'Update phase status',
    {
      phase: z.enum(['INIT', 'SPEC', 'PLAN', 'EXEC', 'REVIEW', 'VERIFY']),
      status: z.enum(['PENDING', 'IN_PROGRESS', 'COMPLETED', 'BLOCKED']),
      basePath: z.string().optional(),
    },
    async ({ phase, status, basePath }) => {
      const manager = new StateManager(basePath || process.cwd());
      const state = manager.updatePhase(phase, { status });
      return {
        content: [{
          type: 'text',
          text: state ? JSON.stringify(state, null, 2) : 'No active session',
        }],
      };
    }
  );

  // kdev_state_confirm - 确认阶段
  server.tool(
    'kdev_state_confirm',
    'Confirm a phase (for phases requiring human approval)',
    {
      phase: z.enum(['SPEC', 'PLAN', 'REVIEW']),
      basePath: z.string().optional(),
    },
    async ({ phase, basePath }) => {
      const manager = new StateManager(basePath || process.cwd());
      const state = manager.confirmPhase(phase, 'user');
      return {
        content: [{
          type: 'text',
          text: state ? `Phase ${phase} confirmed.\n\n${JSON.stringify(state, null, 2)}` : 'No active session',
        }],
      };
    }
  );
}

Create: `src/gate/enforcer.ts`

```typescript
import { exec } from 'child_process';
import { promisify } from 'util';
import * as fs from 'fs';
import * as path from 'path';
import { GateCheck, GateResult, GateType, GATE_CHECKS } from './types.js';

const execAsync = promisify(exec);

export class GateEnforcer {
  private basePath: string;
  private kdevPath: string;

  constructor(basePath: string) {
    this.basePath = basePath;
    this.kdevPath = path.join(basePath, '.kdev');
  }

  /**
   * 获取阶段的所有门禁检查
   */
  getChecksForPhase(phase: string): GateCheck[] {
    return GATE_CHECKS[phase] || [];
  }

  /**
   * 执行单个门禁检查
   */
  async check(check: GateCheck): Promise<GateResult> {
    const result: GateResult = {
      type: check.type,
      passed: false,
      timestamp: new Date().toISOString(),
    };

    try {
      if (check.type === 'SPEC_COMPLETENESS') {
        result.passed = this.checkSpecCompleteness();
      } else if (check.type === 'PLAN_COMPLETENESS') {
        result.passed = this.checkPlanCompleteness();
      } else if (check.type === 'CODE_REVIEW') {
        // 代码审查需要人工确认
        const reviewPath = path.join(this.kdevPath, 'reviews', 'latest.json');
        result.passed = fs.existsSync(reviewPath);
      } else if (check.checkCommand) {
        const { stdout, stderr } = await execAsync(check.checkCommand, {
          cwd: this.basePath,
        });
        result.passed = true;
        result.output = stdout;
      }
    } catch (error) {
      result.passed = false;
      result.error = String(error);
    }

    return result;
  }

  /**
   * 执行阶段所有门禁检查
   */
  async checkPhase(phase: string): Promise<{ allPassed: boolean; results: GateResult[] }> {
    const checks = this.getChecksForPhase(phase);
    const results: GateResult[] = [];

    for (const check of checks) {
      const result = await this.check(check);
      results.push(result);
    }

    const allPassed = results.every((r) => r.passed);
    return { allPassed, results };
  }

  /**
   * 检查 Spec 完整性
   */
  private checkSpecCompleteness(): boolean {
    const specPath = path.join(this.kdevPath, 'specs', 'current.md');
    if (!fs.existsSync(specPath)) return false;

    const content = fs.readFileSync(specPath, 'utf-8');
    const requiredSections = ['# Feature', '## Requirements', '## Success Criteria'];
    return requiredSections.every((s) => content.includes(s));
  }

  /**
   * 检查 Plan 完整性
   */
  private checkPlanCompleteness(): boolean {
    const planPath = path.join(this.kdevPath, 'plans', 'current.md');
    if (!fs.existsSync(planPath)) return false;

    const content = fs.readFileSync(planPath, 'utf-8');
    return content.includes('### Task');
  }
}
```
