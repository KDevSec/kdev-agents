### 1.2 GraphStore 核心设计

```python
# graph.py 核心特性

class GraphStore:
    """SQLite-backed code knowledge graph."""

    def __init__(self, db_path: str | Path) -> None:
        self._conn = sqlite3.connect(
            str(self.db_path), timeout=30, check_same_thread=False,
            isolation_level=None,  # 禁用隐式事务
        )
        self._conn.execute("PRAGMA journal_mode=WAL")  # WAL 模式
        self._conn.execute("PRAGMA busy_timeout=5000")
        self._init_schema()
        run_migrations(self._conn)  # schema 迁移机制

    # 核心写入方法
    def upsert_node(self, node: NodeInfo, file_hash: str = "") -> int:
        """INSERT OR UPDATE，返回 node ID"""

    def upsert_edge(self, edge: EdgeInfo) -> int:
        """插入边，支持 confidence + confidence_tier"""

    def store_file_batch(self, batch: list[tuple]) -> None:
        """批量原子更新，单事务"""
```

**关键 Schema**：

```sql
-- nodes 表
CREATE TABLE nodes (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    kind TEXT NOT NULL,          -- File, Class, Function, Type, Test
    name TEXT NOT NULL,
    qualified_name TEXT NOT NULL UNIQUE,  -- 唯一标识
    file_path TEXT NOT NULL,
    line_start INTEGER,
    line_end INTEGER,
    language TEXT,
    parent_name TEXT,
    params TEXT,
    return_type TEXT,
    modifiers TEXT,
    is_test INTEGER DEFAULT 0,
    file_hash TEXT,
    extra TEXT DEFAULT '{}',     -- JSON 扩展属性
    updated_at REAL NOT NULL
);

-- edges 表
CREATE TABLE edges (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    kind TEXT NOT NULL,           -- CALLS, IMPORTS_FROM, INHERITS, REFERENCES
    source_qualified TEXT NOT NULL,
    target_qualified TEXT NOT NULL,
    file_path TEXT NOT NULL,
    line INTEGER DEFAULT 0,
    extra TEXT DEFAULT '{}',
    confidence REAL DEFAULT 1.0,  -- 信心评分 0.0~1.0
    confidence_tier TEXT DEFAULT 'EXTRACTED',  -- EXTRACTED/INFERRED/AMBIGUOUS
    updated_at REAL NOT NULL
);
```

## 二、graphify 深度分析

### 2.1 架构设计

```
Pipeline: detect() → extract() → build_graph() → cluster() → analyze() → report() → export()
```

每个阶段是单一函数，通过 dict 和 NetworkX 图通信，无共享状态。

### 2.2 Extraction Schema

```json
{
  "nodes": [

  "edges": [
    {"source": "id_a", "target": "id_b", "relation": "calls|imports|uses|...", "confidence": "EXTRACTED|INFERRED|AMBIGUOUS"}
  ]
}
```

### 2.3 MCP Server 设计

```python
# serve.py

def serve(graph_path: str = "graphify-out/graph.json") -> None:
    G = _load_graph(graph_path)
    server = Server("graphify")

    @server.list_tool()
    async def list_tools() -> list[types.Tool]:
        return [
            types.Tool(name="query_graph", ...),      # BFS/DFS 搜索
            types.Tool(name="get_node", ...),         # 获取节点详情
            types.Tool(name="get_neighbors", ...),    # 获取邻居
            types.Tool(name="get_community", ...),    # 获取社区
            types.Tool(name="god_nodes", ...),        # 高连接节点
            types.Tool(name="graph_stats", ...),      # 图统计
            types.Tool(name="shortest_path", ...),    # 最短路径
        ]
```

### 2.4 Confidence 标签体系

| Label | 含义 |
|-------|------|
| `EXTRACTED` | 关系在源码中明确声明（import、直接调用） |
| `INFERRED` | 合理推断（调用图二次传递、上下文共现） |
| `AMBIGUOUS` | 不确定，需要人工审核 |

### 2.5 多模态支持

graphify 支持：
- **代码**: tree-sitter 解析（同 code-review-graph）
- **文档**: Markdown/文本提取
- **论文**: PDF 解析
- **图片**: 设计图、截图解析
- **视频**: Whisper 转录

### 继承扩展模式

```python
from code_review_graph.graph import GraphStore, GraphNode, GraphEdge
from code_review_graph.parser import NodeInfo, EdgeInfo

class SemanticGraphStore(GraphStore):
    """扩展支持 doc/image 节点"""

    def upsert_doc_node(self, doc: DocNode) -> int:
        node = NodeInfo(
            kind="doc",
            name=doc.name,
            file_path=doc.file_path,
            line_start=0,
            line_end=0,
            extra={"doc_type": doc.doc_type, "doc_concepts": doc.concepts}
        )
        return self.upsert_node(node)

    def upsert_semantic_edge(self, source: str, target: str, confidence: float, reason: str) -> int:
        edge = EdgeInfo(
            kind="SEMANTIC_LINK",
            source=source,
            target=target,
            file_path="",
            extra={"confidence": confidence, "confidence_tier": "INFERRED", "reason": reason}
        )
        return self.upsert_edge(edge)
```
