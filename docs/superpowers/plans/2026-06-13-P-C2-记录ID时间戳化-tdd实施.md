# step_id.py — 顶部加 import
import datetime

def _now_stamp(when: "datetime.datetime | None" = None) -> str:
    """本地秒级时间戳 YYYYMMDD-HHMMSS（注入 when 便于测试）。"""
    when = when or datetime.datetime.now()
    return when.strftime("%Y%m%d-%H%M%S")

def _who_suffix() -> str:
    """git email local-part → sanitized slug；无 git/email → ''（调用方据此省略后缀，绝不写 None）。"""
    email = _git_query("config", "user.email")
    if not email:
        return ""
    local = email.split("@", 1)[0]
    s = _sanitize_slug(local)
    return "" if s == "unknown" else s
```

# step_id.py —— 复用现有 _flock_exclusive/_flock_release（已在文件内）
def _dup_index(rec_type: str, base: str, state_dir: Path) -> int:
    """同 (type, base=<ts>-<who>) 的去重序号；首个=0，之后 atomic 递增。

    用一个 per-base counter 文件 + flock；base 含秒级 ts → 文件量极少且短命。
    """
    state_dir.mkdir(parents=True, exist_ok=True)
    safe = _sanitize_slug(f"{rec_type}-{base}")
    p = state_dir / f"dupidx-{safe}.txt"
    fd = os.open(str(p), os.O_RDWR | os.O_CREAT, 0o644)
    try:
        _flock_exclusive(fd)
        os.lseek(fd, 0, os.SEEK_SET)
        raw = os.read(fd, 64).decode("utf-8", errors="replace").strip()
        cur = int(raw) if raw.isdigit() else 0
        os.lseek(fd, 0, os.SEEK_SET); os.ftruncate(fd, 0)
        os.write(fd, f"{cur + 1}\n".encode("utf-8")); os.fsync(fd)
        return cur
    finally:
        _flock_release(fd); os.close(fd)
```

# step_id.py
RECORD_TYPES = ("Step", "Q", "G", "R", "F")

def mint_record_id(rec_type: str, state_dir: Path,
                   when: "datetime.datetime | None" = None,
                   who: "str | None" = None) -> str:
    """新原语 minting：'Step <ts>-<who>[.n]'（全类型空格分隔）。

    who=None → 自动取 git email 前缀；'' → 省略后缀（no-git 不写 None）。
    """
    if rec_type not in RECORD_TYPES:
        raise ValueError(f"unknown record type: {rec_type}")
    ts = _now_stamp(when)
    who = _who_suffix() if who is None else who
    base = f"{ts}-{who}" if who else ts
    n = _dup_index(rec_type, base, state_dir)
    core = base if n == 0 else f"{base}.{n}"
    return f"{rec_type} {core}"
```

# step_id.py
_TS = r"\d{8}-\d{6}"  # YYYYMMDD-HHMMSS
# 新形：<Type> <ts>[-<who>][.<n>]（空格分隔）
_RE_NEW = re.compile(rf"^(Step|Q|G|R|F)\s+({_TS}(?:-[\w.\-]+)?)$")
# 旧形：Step <slug>-<N>（空格） 或 Q/G/R/F-<N>（连字符）
_RE_OLD_STEP = re.compile(r"^(Step)\s+([\w\-\.]+-\d+)$")
_RE_OLD_QGRF = re.compile(r"^(Q|G|R|F)-(\d+)$")

def parse_record_id(label: str) -> "dict | None":
    """解析记录 ID label（不含 '## ' 前缀、不含标题）。双认旧/新形，未识别 → None。"""
    label = label.strip()
    m = _RE_NEW.match(label)
    if m:
        return {"type": m.group(1), "id": m.group(2), "scheme": "timestamp"}
    m = _RE_OLD_STEP.match(label)
    if m:
        return {"type": m.group(1), "id": m.group(2), "scheme": "legacy"}
    m = _RE_OLD_QGRF.match(label)
    if m:
        return {"type": m.group(1), "id": m.group(2), "scheme": "legacy"}
    return None
```
